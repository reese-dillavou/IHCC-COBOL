Variable definitions
	AlphaNumaric - I-REC, I-ID, I-LNAME, I-NAME, I-FNAME, I-INIT, O-DATE, O-PCTR,O-GPA, O-START-SALARY, O-ID, O-LNAME, O-FNAME, O-SCTR, EOF = "TRUE"
	Numaric I-GPA, I-START-SALARY, I-DATE, C-SCTR=0, C-PCTR=0

MAIN
	CALL INIT
	WHILE EOF="TRUE"
		CALL MAINLINE
	CALL CLOSING
MAINLINE
	CALL CALC
	CALL OUTPUT
	CALL READ
CLOSING
		PRINT BLANK LINE
		O-SCTR = C-SCTR
		PRINT TOTAL LINE
		CLOSE INPUT FILE
		CLOSE OUTPUT FILE
		
CALC
	C-SCTR +=1
	
OUTPUT
	O-ID = I-ID
	O-LNAME = I-LNAME
	O-FNAME = I-FNAME
	O-GPA = I-GPA
	O-START-SALARY = I--START-SALARY
	PRINT DETAIL LINE
	PRINT BLANK LINE
	IF END OF PAGE
		CALL HEADING
INIT
	OPEN INPUT FILE
	OPEN OUTPUT FILE
	REQUEST SYSTEM DATE
	GET I-DATE
	O-DATE = I-DATE
	CALL READ
	CALL HDGS
	
READ
	READ RECORD, I-REC, I-ID, I-NAME, I-LNAME, I-FNAME, I-INIT, I-GPA, I-START-SALARY
	IF END OF FILE
		EOF = "NO"
		
HDGS
	C-PCTR +=1
	O-PCTR = C-PCTR
	PRINT TITLE LINE
	PRINT BLANK LINE
	PRINT COLUMN HEADING LINE 1
	PRINT COLUMN HEADING LINE 2
	PRINT BLANK LINE
